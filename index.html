<!DOCTYPE html>

<html>

  <head>

    <!-- Load Google Charts library -->

    <script type="text/javascript" src="https://www.gstatic.com/charts/loader.js"></script>

    <script type="text/javascript">

      // Load the corechart package and set callback

      google.charts.load('current', {'packages':['corechart']});

      google.charts.setOnLoadCallback(drawChart);


      function drawChart() {

        // Prepare DataTable with Date, Rate, and tooltip

        var data = new google.visualization.DataTable();

        data.addColumn('string', 'Date');

        data.addColumn('number', 'USD/ZAR');

        data.addColumn({type: 'string', role: 'tooltip'});

        data.addRows([

          ['8 Feb 2025',  18.4153, '8 Feb 2025: 18.4153'],

          ['9 Feb 2025',  18.5303, '9 Feb 2025: 18.5303'],

          ['10 Feb 2025', 18.4519, '10 Feb 2025: 18.4519'],

          ['11 Feb 2025', 18.5090, '11 Feb 2025: 18.5090'],

          ['12 Feb 2025', 18.5282, '12 Feb 2025: 18.5282'],

          ['13 Feb 2025', 18.4964, '13 Feb 2025: 18.4964'],

          ['14 Feb 2025', 18.3732, '14 Feb 2025: 18.3732'],

          ['15 Feb 2025', 18.3718, '15 Feb 2025: 18.3718'],

          ['16 Feb 2025', 18.3702, '16 Feb 2025: 18.3702']

        ]);


        // Chart options: clean layout, no legend, with axis titles

        var options = {

          title: 'USD to ZAR (8–16 Feb 2025)',

          hAxis: { title: 'Date' },

          vAxis: { title: 'USD/ZAR Rate' },

          legend: 'none'

        };


        // Draw the line chart in the target div

        var chart = new google.visualization.LineChart(document.getElementById('chart_div'));

        chart.draw(data, options);

      }

    </script>

  </head>

  <body>

    <!-- Chart will be rendered here -->

    <div id="chart_div" style="width:700px; height:400px;"></div>

  </body>

</html>
